# 四、插件开发场景模板


## 4.1 保存前取值校验

```csharp
public override void BeforeSave(BeforeSaveEventArgs e)
{
    // 单据头 - 直接取值
    string billNo = this.EntityObject["FBILLNO"].ToString();

    // 基础资料 - as DynamicObject
    DynamicObject material = this.EntityObject["FMaterialId"] as DynamicObject;
    if (material != null)
    {
        string materialNumber = material["FNUMBER"].ToString();
    }

    // 单据体 - 遍历
    DynamicObjectCollection entries = this.EntityObject["FSaleOrderEntry"] as DynamicObjectCollection;
    foreach (DynamicObject row in entries)
    {
        decimal qty = Convert.ToDecimal(row["FQTY"]);
        string mat = (row["FMaterialId"] as DynamicObject)?["FNUMBER"].ToString();
    }
}
```
