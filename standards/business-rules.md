## 3. 数据验证

### 必填字段验证

```csharp
// ✅ 统一验证方法
public ValidationResult ValidateRequired(DynamicObject obj, string[] requiredFields)
{
    var result = new ValidationResult();

    foreach (var field in requiredFields)
    {
        if (obj[field] == null || string.IsNullOrEmpty(obj[field].ToString()))
        {
            result.AddError(field, $"{field} 不能为空");
        }
    }

    return result;
}
```

### 业务逻辑验证

```csharp
// ✅ 验证业务规则
public bool ValidateBusinessRule(DynamicObject bill)
{
    // 示例：金额必须大于 0
    decimal amount = bill.GetDecimal("amount");
    if (amount <= 0)
    {
        throw new BusinessException("金额必须大于 0");
    }

    // 示例：日期不能早于今天
    DateTime billDate = bill.GetDateTime("billdate");
    if (billDate < DateTime.Today)
    {
        throw new BusinessException("单据日期不能早于今天");
    }

    return true;
}
```
